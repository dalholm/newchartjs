API Reference

// Create a 3D network ball
const ball = NewChart.create('#el', {
  type: 'networkball',
  data: { nodeCount: 40 }
});

// Push event — spawns a cursor
ball.pushEvent();
ball.pushEvent({ label: 'thinking...' });

// Multiple events run in parallel
ball.burst(3, 200);

// Continuous auto-events
ball.startAutoEvents(1500);
ball.stopAutoEvents();
